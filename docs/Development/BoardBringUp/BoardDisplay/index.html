while (1)
{
  color.hue += 1; // Increment hue color space by 1


  RgbColor rgb = HsvToRgb(color);
  uint8_t *i = (uint8_t*)FRAMEBUFFER_BASE;
  while(i < (uint8_t*)(FRAMEBUFFER_BASE + (WIDTH*HEIGHT*BYTES_PER_PIXEL)))
  {
    *i++ = (uint8_t)rgb.bblue; // Write blue color and increment pointer by one byte
    *i++ = (uint8_t)rgb.green; // Write green color and increment pointer by one byte
    *i++ = (uint8_t)rgb.red;   // Write red color and increment pointer by one byte
  }
}
